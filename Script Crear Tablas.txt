CREATE TABLE Empleados (
  Legajo INT PRIMARY KEY,
  Nombre VARCHAR (255),
  Apellido VARCHAR (255),
  Mail VARCHAR (255),
  telefono INTEGER,
  contactoDeEmergencia INTEGER,
  RelacionDeEmpleo VARCHAR (255),
  IdCategoriaEmpleados INT

);


CREATE TABLE CategoriaEmpleados (
  idCatEmp INT PRIMARY key,
  nombreCat varchar (250)
);

CREATE TABLE Licencia (
  IdLic INT PRIMARY KEY,
  NombreLic VARCHAR (255),
  IdCategoriaLicencias INT,
  Dias INT,
  IdCategoriaEstado INT
);

CREATE TABLE CategoriaLicencias (
  idCatLic INT PRIMARY key,
  nombreCatLice varchar (250)
);

CREATE TABLE Estados (
  idCatEst INT PRIMARY key,
  nombreEstado varchar (250)
);
